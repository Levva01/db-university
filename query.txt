SELECT * 
FROM `students`
WHERE YEAR(`date_of_birth`) = 1990;

SELECT *
FROM `courses`
WHERE `cfu` > 10;

SELECT *
FROM `students`
WHERE (DATEDIFF(CURRENT_DATE(), `date_of_birth`) / 365) > 30;

SELECT *
FROM `courses`
WHERE `period` = 'I semestre' 
AND `year` = '1';

SELECT *
FROM `exams`
WHERE `date` = '2020-06-20'
AND HOUR(`hour`) > 13;

SELECT *
FROM `degrees`
WHERE `level` = 'magistrale';

SELECT COUNT(`id`) AS 'totale_dipartimenti'
FROM `departments`;
